##Front End Install Bootstrap using this command
npm install bootstrap

then add this into index.js

// Bootstrap CSS
import "bootstrap/dist/css/bootstrap.min.css";
// Bootstrap Bundle JS
import "bootstrap/dist/js/bootstrap.bundle.min";
